     # File lib/zip/zip_entry.rb, line 481
481:     def get_input_stream(&aProc)
482:       if @ftype == :directory
483:           return yield(NullInputStream.instance) if block_given?
484:           return NullInputStream.instance
485:       elsif @filepath
486:         case @ftype

489:         when :symlink
