 9:       def diff_as_string(data_new, data_old)
10:         data_old = data_old.split(/\n/).map! { |e| e.chomp }
11:         data_new = data_new.split(/\n/).map! { |e| e.chomp }
12:         diffs = Diff::LCS.diff(data_old, data_new)
13:         output = ""
14:         return output if diffs.empty?
15:         oldhunk = hunk = nil
16:         file_length_difference = 0
17:         diffs.each do |piece|
18:           begin
19:             hunk = Diff::LCS::Hunk.new(
20:               data_old, data_new, piece, context_lines, file_length_difference
21:             )
22:             file_length_difference = hunk.file_length_difference
23:             next unless oldhunk
24:             # Hunks may overlap, which is why we need to be careful when our
25:             # diff includes lines of context. Otherwise, we might print
26:             # redundant lines.
27:             if (context_lines > 0) and hunk.overlaps?(oldhunk)
28:               hunk.unshift(oldhunk)
29:             else
30:               output << oldhunk.diff(format)
31:             end
32:           ensure
33:             oldhunk = hunk
34:             output << "\n"
35:           end
36:         end
