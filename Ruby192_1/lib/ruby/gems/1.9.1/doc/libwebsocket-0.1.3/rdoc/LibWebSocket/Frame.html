LibWebSocket::Frame

Construct or parse a WebSocket frame.

SYNOPSIS 

  # Create frame
  frame = LibWebSocket::Frame.new('123')
  frame.to_s # \x00123\xff

  # Parse frames
  frame = LibWebSocket::Frame.new
  frame.append("123\x00foo\xff56\x00bar\xff789")
  frame.next # =>  foo
  frame.next # =>  bar
